### 문제
https://www.acmicpc.net/problem/1012

### 코드(본인작성)

```java
import java.io.*;
import java.util.*;


public class Main {
    static int [][] map;
    static boolean[][] checkList;

    static int[] findX = {0,0,-1,1};
    static int[] findY = {-1,1,0,0};
    static int count;
    static int m;
    static int n;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st;


        st = new StringTokenizer(br.readLine());
        int t = Integer.parseInt(st.nextToken());
        StringBuilder sb = new StringBuilder();
        for (int l = 0; l < t; l++) {
            st = new StringTokenizer(br.readLine());
            m = Integer.parseInt(st.nextToken());
            n = Integer.parseInt(st.nextToken());
            int k = Integer.parseInt(st.nextToken());

            map = new int[m][n];
            checkList = new boolean[m][n];
            for (int i = 0; i < k; i++) {
                st = new StringTokenizer(br.readLine());
                int x = Integer.parseInt(st.nextToken());
                int y = Integer.parseInt(st.nextToken());
                map[x][y] = 1;
            }

            count = 0;

            for (int i = 0; i < m; i++) {
                for (int j = 0; j < n; j++) {
                    if(!checkList[i][j]&&map[i][j]==1){
                        bfs(i,j);
                        count++;
                    }
                    else{
                        continue;
                    }
                }
            }
            sb.append(count+"\n");
        }

        System.out.println(sb.toString());
    }

    private static void bfs(int startX,int startY) {
        checkList[startX][startY] = true;
        LinkedList<Node> q = new LinkedList<>();
        q.add(new Node(startX,startY));

        while(!q.isEmpty()){
            Node node = q.poll();

            for (int i = 0; i < 4; i++) {
                int x1 = node.x+findX[i];
                int y1 = node.y+findY[i];
                if(x1>=0&&y1>=0&&y1<n&&x1<m){
                    if(map[x1][y1]==1&&!checkList[node.x+findX[i]][node.y+findY[i]])
                    {
                        checkList[x1][y1] = true;
                        q.add(new Node(x1,y1));
                    }
                }
            }

        }
    }

    static class Node{
        int x;
        int y;

        public Node(int x, int y) {
            this.x = x;
            this.y = y;
        }
    }

}





```
