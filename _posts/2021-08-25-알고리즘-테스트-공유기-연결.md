### 코드(본인작성)

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.*;


public class Main {


    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st;
        StringBuilder sb;

        st = new StringTokenizer(br.readLine());

        int homeCount = Integer.parseInt(st.nextToken());
        int router =  Integer.parseInt(st.nextToken());

        int[] home = new int[homeCount];
        for (int i = 0; i < homeCount; i++) {
            home[i] = Integer.parseInt(br.readLine());
        }

        Arrays.sort(home);

        int high = home[home.length-1]-home[0];
        int low = 1;
        int mid = 0;
        int answer=0;
        while(high>=low){
            mid = (high+low)/2;

            int count=0;
            for (int i = 0; i < home.length-1; i++) {
                for (int j = i+1; j < home.length; j++) {
                    if(home[j]-home[i]>=mid){
                        count++;
                        i = j-1;
                        break;
                    }
                }
            }

            if(count>=router-1){
                answer = mid;
                low = mid +1;
            }
            else{
                high = mid-1;
            }
        }

        System.out.println(answer);
    }
}



```
